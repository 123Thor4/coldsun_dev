	desc = artifact.1.d

		if = {
			limit = {
				has_country_flag = artifact_one
			}
			clr_country_flag = artifact_one

			limit = {

		}

		name = artifact.1.a

		name = artifact.1.b
		set_country_flag = artifact_two
	}
	option = {
		name = artifact.1.c

	}
}
