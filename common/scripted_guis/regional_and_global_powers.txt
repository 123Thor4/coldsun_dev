scripted_gui = {
	powers_top_button = {
		context_type = player_context
		
		window_name = "powers_container"
		
		visible = {
		}
		
		effects = {
			powers_button_click = {
				set_variable = {
					powers_show = 1
				}
			}
		}
	}
	
	powers_window_thing = {
		context_type = player_context
		
		window_name = "powers_window"
		
		visible = {
			check_variable = {
				powers_show > 0
			}
		}
		
		effects = {
			exitbutton_powers_click = {
				set_variable = {
					powers_show = 0
				}
			}
		}
	}
}
