# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = { # Luna's Anger
	name = getlunasanger
	text = {
		trigger = {
			check_variable = {
				var = luna_anger
				val < 19
			}
		}
		localization_key = lunas_anger_1
	}
	
	text = {
		trigger = {
			check_variable = {
				var = luna_anger
				val > 19
			}
			check_variable = {
				var = luna_anger
				val < 39
			}
		}
		localization_key = lunas_anger_2
	}
	
	text = {
		trigger = {
			check_variable = {
				var = luna_anger
				val > 39
			}
			check_variable = {
				var = luna_anger
				val < 59
			}
		}
		localization_key = lunas_anger_3
	}
	
	text = {
		trigger = {
			check_variable = {
				var = luna_anger
				val > 59
			}
			check_variable = {
				var = luna_anger
				val < 79
			}
		}
		localization_key = lunas_anger_4
	}
	
	text = {
		trigger = {
			check_variable = {
				var = luna_anger
				val > 79
			}
		}
		localization_key = lunas_anger_5
	}

}
